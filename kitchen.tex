\centering Rice in Dutch oven
\begin{description}[noitemsep]
 \item[Preheat] oven to {\color{red}375{\textdegree}F}
 \item[Rinse] 1 cup {\color{blue}brown rice}
 \item[Boil] 2 $\sfrac{1}{2}$ cups {\color{blue}water}, $\sfrac{1}{2}$ tbsp {\color{blue}butter}, 1 tsp {\color{blue}salt}.
 \item[Bake] Add rice, cover, bake for 55 minutes (25 minutes for white rice).
 \item[Fluff] let the steam out with a fork.
\end{description}

\noindent\fbox{
\begin{minipage}[t]{5.75in}
\centering Whole-Wheat Honey Bread
\begin{description}[noitemsep]
\end{description}
  \item[Sponge] 1 cup {\color{blue}bread flour}, $\sfrac{1}{4}$ cup {\color{blue}honey}, 2 tsp {\color{blue}yeast}, 3 cups {\color{blue}warm (110-115{\textdegree}F) water}.  Let sit about 5 minutes, until foamy.
  \item[Add] 2 cups {\color{blue}bread flour}, 3 cups {\color{blue}whole wheat flour}, $\sfrac{1}{2}$ cup {\color{blue}flax seeds}.
  \item[Mix] in stand mixer on low speed until dough comes together in shaggy mass, about 1 minute.  Add 1 tbsp {\color{blue}salt}, 1 tbsp {\color{blue}melted butter}.  Mix on medium speed until dough is smooth, about 5 minutes.
  \item[Knead] about 5 minutes by hand.  Put back in oiled bowl, cover, for about 45 minutes.
  \item[Final Rise] pull out a bit of dough.  Divide remaining dough in two, put in loaf pans, cover, for about 30 minutes.
  \item[Preheat] oven to {\color{red}375{\textdegree}F}
  \item[Bake] score loaves, apply 2 tbsp {\color{blue}melted butter}, bake for 35 to 45 minutes.
\end{description}
\end{minipage}
